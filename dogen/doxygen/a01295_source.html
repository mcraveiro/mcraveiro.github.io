169  return false;

173  //note: this does accept long int double --> I'll this in the parser
174  if (type_name.empty())
175  {
176  type_name = "int";
177  return true;
178  }
179  else if ((type_name == "long") || (type_name == "long long") ||
180  (type_name == "short"))
181  return true;
182  else
183  return false;
184  }
185  bool set_long()
186  {
187  if (type_name.empty() || (type_name == "int"))
188  type_name = "long";
189  else if (type_name == "long")
190  type_name = "long long";
191  else if (type_name == "double")
192  type_name = "long double";
193  else
194  return false;
195  return true;
196  }
197 
198  bool set_double() //because "long double"
199  {
200  if (type_name.empty())
201  type_name = "double";
202  else if (type_name == "long")
203  type_name = "long double";
204  else
205  return false;
206  return true;
207  }
208 
209  bool set_reference()
210  {

226  }

304  static bool is_keyword(const std::string & input)
305  {
306  const static std::vector<const char*> keywords =
307  {
308  "alignas", "alignof", "and", "and_eq", "asm", "auto", "bitand", "bitor", "bool", "break",
309  "case", "catch", "char", "char16_t", "char32_t", "class", "compl", "const", "constexpr",
310  "const_cast", "continue", "decltype", "default", "delete", "do", "double", "dynamic_cast",
311  "else", "enum", "explicit", "export", "extern", "false", "float", "for", "friend", "goto",
312  "if", "inline", "int", "long", "mutable", "namespace", "new", "noexcept", "not", "not_eq",
313  "nullptr", "operator", "or", "or_eq", "private", "protected", "public", "register", "reinterpret_cast",
314  "return", "short", "signed", "sizeof", "static", "static_assert", "static_cast", "struct", "switch",
315  "template", "this", "thread_local", "throw", "true", "try", "typedef", "typeid", "typename", "union",
316  "unsigned", "using", "virtual", "void", "volatile", "wchar_t", "while", "xor", "xor_eq"
317  };
318  return std::count_if(keywords.begin(), keywords.end(), [&](const char * st){return st == input;}) != 0;
