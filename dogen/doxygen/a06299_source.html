77 class indent_filter : public boost::iostreams::output_filter {
78 private:
79  void reset_blank_lines_management() {
80  manage_blank_lines_ = false;
81  found_cr_ = false;
82  found_lf_ = false;
83  }
84 
85 public:
86  template<typename Sink>
87  bool put(Sink& dest, int i) {
88  const char c(static_cast<char>(i));
89  const char cr(0x0D);
90  const char lf(0x0A);
91 
92  if (manage_blank_lines_) {
93  if (c == cr) {
94  found_cr_ = true;
95  return true;
96  } else if (c == lf) {
97  found_lf_ = true;
98  return true;
99  }
100 
101  if (!is_first_character_) {
102  if (found_cr_) {
103  found_cr_ = false;
104  boost::iostreams::put(dest, cr);
105  }
106 
107  if (found_lf_) {
108  found_lf_ = false;
109  boost::iostreams::put(dest, lf);
110  }
111  }
112  reset_blank_lines_management();
113  }
114  is_first_character_ = false;
115 
116  if (c == lf) {
117  at_line_start_ = true;
118  return boost::iostreams::put(dest, c);
119  }
120 
121  if (at_line_start_) {
122  at_line_start_ = false;
123  const auto total_spaces(indentation_size_ * indentation_level_);
124  for (unsigned int n(0); n < total_spaces; ++n)
125  boost::iostreams::put(dest, ' ');
126  }
127  return boost::iostreams::put(dest, c);
128  }

130  template<typename Sink>
131  void close(Sink&) {
132  indentation_level_ = 0;
133  at_line_start_ = true;
134  }

145  void decrement_indentation_level(const bool reset_management = true) {
146  if (indentation_level_ > 0) {
147  --indentation_level_;

161  static void push(boost::iostreams::filtering_ostream& s,

176  }

190  }

194  : indentation_level_(0), at_line_start_(true), is_first_character_(true),
195  indentation_size_(indentation_size), stream_(0),
